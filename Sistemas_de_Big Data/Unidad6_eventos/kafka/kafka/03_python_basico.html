🐍 Producer y Consumer en Python

Tu primer producer y consumer con confluent-kafka. Configuración correcta, manejo de errores, offset management y patrones para producción. Código listo para ejecutar.

🏭 Producer — Escribir mensajes
Producer mínimo — funcional pero incompleto
from confluent_kafka import Producer

# Configuración básica
conf = {'bootstrap.servers': 'localhost:9094'}
producer = Producer(conf)

# Callback: llamado cuando el mensaje llega al broker (o falla)
def delivery_report(err, msg):
    if err is not None:
        print(f'❌ Error: {err}')
    else:
        print(f'✅ Entregado → topic={msg.topic()} partition={msg.partition()} offset={msg.offset()}')

# Publicar mensajes
for i in range(10):
    producer.produce(
        topic='mi-topic',
        key=f'user_{i % 3}',       # clave → determina la partición
        value=f'mensaje número {i}',
        callback=delivery_report
    )
    producer.poll(0)  # dispara callbacks pendientes sin bloquear

# IMPORTANTE: flush espera a que todos los mensajes sean enviados
producer.flush()
print('Todos los mensajes enviados')
⚠️ poll(0) y flush() son críticos. Sin flush(), el programa puede terminar antes de enviar los mensajes. poll() dispara callbacks de entrega.
Producer de producción — con JSON, métricas y configuración robusta
from confluent_kafka import Producer
import json, logging, time
from datetime import datetime

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger('kafka-producer')

# ─── Configuración de producción ───────────────────────────────────────────
conf = {
    'bootstrap.servers': 'localhost:9094',
    'acks': 'all',           # esperar confirmación de TODAS las réplicas in-sync
    'enable.idempotence': True, # garantiza exactly-once desde el producer
    'max.in.flight.requests.per.connection': 5,
    'retries': 2147483647,   # reintentos máximos
    'retry.backoff.ms': 100,
    'linger.ms': 5,          # esperar 5ms para batching (más throughput)
    'batch.size': 65536,     # 64KB por batch
    'compression.type': 'snappy', # compresión (snappy, lz4, gzip, zstd)
}

producer = Producer(conf)

# ─── Esquema de mensaje de evento ──────────────────────────────────────────
def crear_evento_pedido(usuario_id: str, items: list, total: float) -> dict:
    return {
        "event_type": "pedido_creado",
        "schema_version": "1.0",
        "timestamp": datetime.utcnow().isoformat() + "Z",
        "usuario_id": usuario_id,
        "items": items,
        "total": total,
    }

# ─── Callback con logging completo ─────────────────────────────────────────
def delivery_report(err, msg):
    if err:
        logger.error(f"❌ Fallo entrega: {err} → topic={msg.topic()} key={msg.key()}")
    else:
        logger.info(
            f"✅ t={msg.topic()} p={msg.partition()} "
             f"off={msg.offset()} key={msg.key().decode()}"
        )

# ─── Función de alto nivel para publicar eventos ────────────────────────────
def publicar_evento(usuario_id: str, items: list, total: float):
    evento = crear_evento_pedido(usuario_id, items, total)
    try:
        producer.produce(
            topic="pedidos",
            key=usuario_id.encode(),          # str → bytes (obligatorio)
            value=json.dumps(evento).encode(), # dict → JSON → bytes
            callback=delivery_report,
        )
        producer.poll(0)
    except BufferError:
        # Cola interna llena → esperar y reintentar
        logger.warning("Producer buffer lleno, esperando...")
        producer.flush()
        producer.produce(...)  # reintentar

# Uso:
publicar_evento("user_123", [{"sku": "A001", "qty": 2}], 59.99)
publicar_evento("user_456", [{"sku": "B002", "qty": 1}], 29.99)
producer.flush()  # siempre al final
📡 Consumer — Leer mensajes
Consumer básico con auto-commit
from confluent_kafka import Consumer, KafkaError, KafkaException

conf = {
    'bootstrap.servers': 'localhost:9094',
    'group.id': 'mi-consumer-group',  # ← obligatorio
    'auto.offset.reset': 'earliest',    # 'earliest' o 'latest'
    'enable.auto.commit': True,          # commit automático cada 5s
}

consumer = Consumer(conf)
consumer.subscribe(['pedidos'])

try:
    while True:
        msg = consumer.poll(timeout=1.0)  # esperar max 1s por mensaje

        if msg is None:
            continue  # timeout, volver a intentar

        if msg.error():
            if msg.error().code() == KafkaError._PARTITION_EOF:
                # Llegamos al final de la partición — normal
                print(f'Fin de partición {msg.partition()} offset {msg.offset()}')
            else:
                raise KafkaException(msg.error())
            continue

        # Procesar el mensaje
        key = msg.key().decode('utf-8') if msg.key() else None
        value = json.loads(msg.value().decode('utf-8'))

        print(f'topic={msg.topic()} partition={msg.partition()} '
              f'offset={msg.offset()} key={key}')
        print(f'Evento: {value}')

finally:
    consumer.close()  # ← siempre cerrar para liberar el group membership
Consumer de producción — commit manual + gestión de errores
from confluent_kafka import Consumer, KafkaError
import json, logging, signal

logger = logging.getLogger('kafka-consumer')
running = True

# Graceful shutdown con Ctrl+C / SIGTERM
def shutdown(signum, frame):
    global running
    running = False

signal.signal(signal.SIGINT, shutdown)
signal.signal(signal.SIGTERM, shutdown)

conf = {
    'bootstrap.servers': 'localhost:9094',
    'group.id': 'pedidos-processor',
    'auto.offset.reset': 'earliest',
    'enable.auto.commit': False,    # ← COMMIT MANUAL (más control)
    'max.poll.interval.ms': 300000, # 5min antes de considerar consumer muerto
    'session.timeout.ms': 45000,    # 45s heartbeat timeout
}

# Callbacks de rebalanceo
def on_assign(consumer, partitions):
    logger.info(f'Particiones asignadas: {[p.partition for p in partitions]}')

def on_revoke(consumer, partitions):
    logger.info(f'Particiones revocadas: {[p.partition for p in partitions]}')
    # ← aquí comittear offsets antes del rebalanceo
    consumer.commit(asynchronous=False)

consumer = Consumer(conf)
consumer.subscribe(['pedidos'], on_assign=on_assign, on_revoke=on_revoke)

def procesar_pedido(evento: dict) -> bool:
    """Devuelve True si el procesamiento fue exitoso."""
    try:
        # Tu lógica de negocio aquí
        logger.info(f"Procesando pedido de {evento['usuario_id']}: {evento['total']}€")
        return True
    except Exception as e:
        logger.error(f"Error procesando: {e}")
        return False

try:
    while running:
        msg = consumer.poll(timeout=1.0)
        if msg is None or msg.error():
            continue

        evento = json.loads(msg.value())

        # Procesar ANTES de commitear el offset
        exito = procesar_pedido(evento)

        if exito:
            # Solo commiteamos si el procesamiento fue exitoso
            # Esto garantiza at-least-once: si falla, releeremos
            consumer.commit(asynchronous=False)
        else:
            # En caso de error: loguear y decidir (DLQ, skip, etc.)
            logger.error(f"Mensaje sin procesar: offset={msg.offset()}")

finally:
    consumer.close()
    logger.info("Consumer cerrado correctamente")
🎯 La regla de oro: Commitear el offset DESPUÉS de procesar, nunca antes. Si commiteamos antes y el procesamiento falla, el mensaje se pierde para siempre. Si commiteamos después, en caso de crash releeremos pero procesaremos de nuevo (at-least-once).
⚡ Consumer Asíncrono con aiokafka
Cuando necesitas integrar Kafka con asyncio
from aiokafka import AIOKafkaConsumer, AIOKafkaProducer
import asyncio, json

# Consumer async — perfecto para FastAPI o código async existente
async def consumir_async():
    consumer = AIOKafkaConsumer(
        "pedidos",
        bootstrap_servers="localhost:9094",
        group_id="async-processor",
        auto_offset_reset="earliest",
        enable_auto_commit=False,
        value_deserializer=lambda v: json.loads(v.decode()),
    )
    await consumer.start()
    try:
        async for msg in consumer:
            print(f'offset={msg.offset} key={msg.key} value={msg.value}')
            await procesar_asincrono(msg.value)
            await consumer.commit()
    finally:
        await consumer.stop()

# Producer async
async def producir_async():
    producer = AIOKafkaProducer(
        bootstrap_servers="localhost:9094",
        value_serializer=lambda v: json.dumps(v).encode(),
    )
    await producer.start()
    try:
        await producer.send_and_wait(
            "pedidos",
            key=b"user_123",
            value={"pedido": 1, "total": 99.9}
        )
    finally:
        await producer.stop()

asyncio.run(consumir_async())
💡 aiokafka es ideal para integrarlo en aplicaciones FastAPI donde ya tienes un event loop. El consumer se puede ejecutar como background task al arrancar la aplicación.
📊 Configuración clave del Producer y Consumer
Parámetros Producer importantes
Parámetro Valor recomendado Por qué
acks 'all' Máxima durabilidad
enable.idempotence True Evita duplicados
linger.ms 5-10 Mejor batching
compression.type snappy Velocidad+compresión
retries alto (2^31) Auto-recuperación
Parámetros Consumer importantes
Parámetro Valor recomendado Por qué
enable.auto.commit False Control manual del offset
auto.offset.reset earliest Procesar desde el inicio
max.poll.records 500 Batch size por poll
fetch.min.bytes 1024 Menos peticiones vacías
session.timeout.ms 45000 Balance latencia/estabilidad
