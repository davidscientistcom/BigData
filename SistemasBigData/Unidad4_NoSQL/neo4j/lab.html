1. Python + Neo4j Driver

python — Driver básico: sesiones y transacciones
from neo4j import GraphDatabase
import os

URI = os.getenv("NEO4J_URI", "bolt://localhost:7687")
AUTH = (os.getenv("NEO4J_USER", "neo4j"), os.getenv("NEO4J_PASS", "neuralbrain"))

class Neo4jClient:
    def __init__(self, uri: str, auth: tuple):
        self.driver = GraphDatabase.driver(uri, auth=auth)
        self.driver.verify_connectivity()
        print("✅ Conexión establecida")

    def close(self):
        self.driver.close()

    # Transacción de lectura (never modifies)
    def get_papers_by_author(self, nombre: str) -> list[dict]:
        with self.driver.session(database="neo4j") as session:
            result = session.execute_read(self._query_papers, nombre)
            return result

    @staticmethod
    def _query_papers(tx, nombre: str):
        query = """
        MATCH (inv:Investigador {nombre: $nombre})-[:PUBLICÓ]->(p:Paper)
        RETURN p.titulo AS titulo, p.citaciones AS citas
        ORDER BY citas DESC
        """
        records = tx.run(query, nombre=nombre)
        return [dict(r) for r in records]

    # Transacción de escritura con manejo de errores
    def upsert_paper(self, doi: str, titulo: str, citaciones: int) -> bool:
        with self.driver.session(database="neo4j") as session:
            try:
                session.execute_write(self._upsert_paper, doi, titulo, citaciones)
                return True
            except Exception as e:
                print(f"❌ Error: {e}")
                return False

    @staticmethod
    def _upsert_paper(tx, doi: str, titulo: str, citaciones: int):
        tx.run(
            "MERGE (p:Paper {doi: $doi}) SET p.titulo = $titulo, p.citaciones = $citaciones",
            doi=doi, titulo=titulo, citaciones=citaciones
        )

# Uso
if __name__ == "__main__":
    client = Neo4jClient(URI, AUTH)
    papers = client.get_papers_by_author("Deivit")
    for p in papers:
        print(f"  {p['titulo']} ({p['citas']} citas)")
    client.close()

2. GraphRAG con LangChain + Neo4j

Construir un sistema de Retrieval-Augmented Generation donde el contexto viene del grafo, no solo de documentos vectoriales.

python — LangChain + Neo4jVector + GraphCypherQA
from langchain_community.graphs import Neo4jGraph
from langchain_community.vectorstores import Neo4jVector
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain.chains import GraphCypherQAChain

# Conexión al grafo
graph = Neo4jGraph(
    url="bolt://localhost:7687",
    username="neo4j",
    password="neuralbrain"
)

# Vector store sobre embeddings de papers
vector_index = Neo4jVector.from_existing_index(
    OpenAIEmbeddings(),
    graph=graph,
    index_name="paper_embedding",
    node_label="Paper",
    text_node_property="abstract",
    embedding_node_property="embedding"
)

# Búsqueda semántica pura
docs = vector_index.similarity_search("LIDAR point cloud processing", k=5)
for d in docs: print(d.page_content[:200])

# GraphCypherQA: el LLM genera Cypher desde lenguaje natural
chain = GraphCypherQAChain.from_llm(
    ChatOpenAI(model="gpt-4o"),
    graph=graph,
    verbose=True,
    allow_dangerous_requests=True  # requerido en v0.3+
)

result = chain.invoke({"query": "¿Qué investigadores colaboran con Deivit en papers sobre LIDAR?"})
print(result["result"])
💡
GraphRAG vs RAG puro: El RAG vectorial encuentra documentos similares semánticamente, pero no puede razonar sobre relaciones. GraphRAG combina: (1) embedding search para recuperar nodos relevantes, (2) traversal del grafo para expandir contexto de relaciones, (3) el LLM sintetiza ambas fuentes. Ideal para knowledge graphs donde las conexiones son el dato más valioso.

4. Proyecto Final: Knowledge Graph de Vehículos Autónomos

🎯
Objetivo: Construir un Knowledge Graph completo que modele accidentes de vehículos autónomos, causas raíz, sensores implicados, condiciones ambientales y decisiones del sistema. 5 fases progresivas.

Fase 1: Schema y Constraints

cypher — Schema del Knowledge Graph AV
-- Constraints (identidad única por entidad)
CREATE CONSTRAINT av_vin        FOR (v:Vehiculo) REQUIRE v.vin IS UNIQUE
CREATE CONSTRAINT acc_id        FOR (a:Accidente) REQUIRE a.id IS UNIQUE
CREATE CONSTRAINT sensor_serial FOR (s:Sensor) REQUIRE s.serial IS UNIQUE
CREATE CONSTRAINT causa_id      FOR (c:CausaRaiz) REQUIRE c.id IS UNIQUE

-- Índices para queries frecuentes
CREATE INDEX acc_severidad FOR (a:Accidente) ON (a.severidad)
CREATE INDEX acc_fecha     FOR (a:Accidente) ON (a.fecha)
CREATE INDEX sensor_tipo   FOR (s:Sensor) ON (s.tipo)

Fase 2: Carga de Datos de Demostración

cypher — Insertar dataset de prueba
-- Vehículos
UNWIND [
  {vin: "AV-001", modelo: "Tesla S", nivel_autonomia: 4},
  {vin: "AV-002", modelo: "Waymo One", nivel_autonomia: 5},
  {vin: "AV-003", modelo: "Cruise AV", nivel_autonomia: 4}
] AS v_data
MERGE (v:Vehiculo {vin: v_data.vin})
SET v.modelo = v_data.modelo, v.nivel_autonomia = v_data.nivel_autonomia

-- Sensores
UNWIND [
  {serial: "S-L01", tipo: "LIDAR", alcance: 200, vin: "AV-001"},
  {serial: "S-R01", tipo: "RADAR", alcance: 150, vin: "AV-001"},
  {serial: "S-C01", tipo: "CAMERA", alcance: 80, vin: "AV-002"}
] AS s_data
MERGE (s:Sensor {serial: s_data.serial})
SET s.tipo = s_data.tipo, s.alcance_m = s_data.alcance
WITH s, s_data
MATCH (v:Vehiculo {vin: s_data.vin})
MERGE (v)-[:EQUIPADO_CON]->(s)

-- Accidentes con causas
MERGE (acc:Accidente {id: "ACC-001"})
SET acc.fecha = date("2026-01-15"), acc.severidad = 3, acc.descripcion = "Colisión trasera en lluvia"
MERGE (causa:CausaRaiz {id: "C-001"})
SET causa.descripcion = "Fallo de detección LIDAR por lluvia intensa"
MERGE (acc)-[:CAUSADO_POR]->(causa)
WITH acc
MATCH (v:Vehiculo {vin: "AV-001"}), (s:Sensor {serial: "S-L01"})
MERGE (v)-[:INVOLUCRADO_EN]->(acc)
MERGE (s)-[:FALLÓ_EN]->(acc)

-- Condición ambiental
MERGE (env:Condicion {id: "ENV-001"})
SET env.lluvia_mm = 25, env.visibilidad_m = 50, env.temperatura = 12
MATCH (acc:Accidente {id: "ACC-001"})
MERGE (acc)-[:OCURRIÓ_BAJO]->(env)

Fase 3: Queries de Negocio

cypher — Análisis de accidentes
-- ¿Qué tipos de sensor fallan más frecuentemente?
MATCH (s:Sensor)-[:FALLÓ_EN]->(acc:Accidente)
RETURN s.tipo, count(acc) AS fallos, avg(acc.severidad) AS severidad_media
ORDER BY fallos DESC

-- ¿Hay correlación entre lluvia y fallos de LIDAR?
MATCH (s:Sensor {tipo: "LIDAR"})-[:FALLÓ_EN]->(acc:Accidente)-[:OCURRIÓ_BAJO]->(env:Condicion)
RETURN env.lluvia_mm, count(acc) AS accidentes
ORDER BY env.lluvia_mm DESC

-- Cadena causal completa de un accidente
MATCH (acc:Accidente {id: "ACC-001"})
OPTIONAL MATCH (acc)-[:CAUSADO_POR]->(causa)
OPTIONAL MATCH (sensor)-[:FALLÓ_EN]->(acc)
OPTIONAL MATCH (vehiculo)-[:INVOLUCRADO_EN]->(acc)
OPTIONAL MATCH (acc)-[:OCURRIÓ_BAJO]->(env)
RETURN acc, causa, sensor, vehiculo, env

Fase 4: Optimización y GDS

cypher — GDS para análisis de red de causas
-- Proyectar red de causalidad
CALL gds.graph.project(
  "causalidad-av",
  ["Accidente", "CausaRaiz", "Sensor", "Condicion"],
  ["CAUSADO_POR", "FALLÓ_EN", "OCURRIÓ_BAJO"]
)

-- Centralidad de causas raíz (las más recurrentes en accidentes)
CALL gds.betweenness.stream("causalidad-av")
YIELD nodeId, score
WITH gds.util.asNode(nodeId) AS nodo, score
WHERE nodo:CausaRaiz
RETURN nodo.descripcion, score ORDER BY score DESC

Fase 5: Integración Python + API

python — FastAPI endpoint para consultas al grafo
from fastapi import FastAPI
from neo4j import GraphDatabase
from pydantic import BaseModel

app = FastAPI(title="AV Knowledge Graph API")
driver = GraphDatabase.driver("bolt://localhost:7687", auth=("neo4j", "neuralbrain"))

class AccidenteQuery(BaseModel):
    min_severidad: int = 1

@app.get("/accidentes")
def get_accidentes(min_severidad: int = 1):
    with driver.session() as session:
        result = session.run(
            """
            MATCH (acc:Accidente)
            WHERE acc.severidad >= $min_sev
            OPTIONAL MATCH (acc)-[:CAUSADO_POR]->(causa)
            RETURN acc.id, acc.severidad, acc.fecha, causa.descripcion
            ORDER BY acc.severidad DESC
            """,
            min_sev=min_severidad
        )
        return [dict(r) for r in result]

@app.get("/sensores/{tipo}/fallos")
def fallos_por_tipo(tipo: str):
    with driver.session() as session:
        result = session.run(
            """
            MATCH (s:Sensor {tipo: $tipo})-[:FALLÓ_EN]->(acc:Accidente)
            RETURN s.serial, count(acc) AS total_fallos, avg(acc.severidad) AS severidad_media
            """,
            tipo=tipo
        )
        return [dict(r) for r in result]
